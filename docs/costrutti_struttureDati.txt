'''
Esempi di costrutti e strutture dati in Python
'''

# Condizionali (if, elif, else):
x = 10
if x > 0:
    print("x è positivo")
elif x == 0:
    print("x è zero")
else:
    print("x è negativo")

# Cicli (for, while):
for elemento in [1, 2, 3, 4, 5]:
    print(elemento)

contatore = 0
while contatore < 5:
    print(contatore)
    contatore += 1

# Funzioni:
def somma(a, b):
    return a + b

risultato = somma(3, 4)
print("La somma è:", risultato)

# Liste:
lista = [1, 2, 3, 4, 5]
lista.append(6)
print("Elementi della lista:", lista)

# Tuple:
tupla = (1, 2, 3)
print("Elementi della tupla:", tupla)

# Dizionari:
dizionario = {'chiave1': 'valore1', 'chiave2': 'valore2'}
print("Valore associato a chiave1:", dizionario['chiave1'])
# Per stampare il valore della chiave: print("Chiavi del dizionario:", dizionario.keys())

# Insiemi:
insieme = {1, 2, 3, 3, 4, 5}
print("Elementi dell'insieme:", insieme)

# Comprehension delle liste:
quadrati = [x**2 for x in range(1, 6)]
print("Quadrati:", quadrati)

# Gestione delle eccezioni (try, except):
try:
    risultato = 10 / 0
except ZeroDivisionError:
    risultato = "Errore di divisione per zero"
print("Risultato:", risultato)

# Lambda function:
quadrato = lambda x: x**2
print("Il quadrato di 4 è:", quadrato(4))
